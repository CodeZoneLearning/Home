equipment_summary.py
def summarize(tests):
    statuses = [test["status"] for test in tests]
    mode = max(set(statuses), key=statuses.count)
    return {
        "tests": len(tests),
        "inferred_status": mode,
        "consistent": len(set(statuses)) == 1,
    }
